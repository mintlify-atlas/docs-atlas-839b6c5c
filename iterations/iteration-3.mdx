### Create MSW handlers for category endpoints

In `src/mocks/handlers.ts`, add handlers for categories and category products:

```typescript
import categoriesFixtures from './categories-fixtures.json'
import productsFixtures from './products-fixtures.json'

export const handlers = [
  http.get('/categories', () => {
    return HttpResponse.json(categoriesFixtures)
  }),
  
  http.get('/products', () => {
    return HttpResponse.json(productsFixtures)
  }),
  
  // Get category by slug
  http.get('/categories/:slug', ({ params }) => {
    const { slug } = params
    const category = categoriesFixtures.find(c => c.slug === slug)
    
    if (!category) {
      return new HttpResponse(null, { status: 404 })
    }
    
    return HttpResponse.json(category)
  }),
  
  // Get products for a category
  http.get('/categories/:slug/products', ({ params }) => {
    const { slug } = params
    const category = categoriesFixtures.find(c => c.slug === slug)
    
    if (!category) {
      return new HttpResponse(null, { status: 404 })
    }
    
    const categoryProducts = productsFixtures.filter(
      p => p.categoryId === category.id
    )
    
    return HttpResponse.json(categoryProducts)
  }),
]
```

### Implement the CategoryDetail component

In `src/pages/category-detail/CategoryDetail.tsx`:

```typescript
import { useEffect, useState } from 'react'
import { useParams } from 'react-router-dom'
import { ProductCard } from '../../components/product-card/ProductCard'
import type { Category, Product } from '../../types'
import './CategoryDetail.css'

const CategoryDetail = () => {
  const { slug } = useParams<{ slug: string }>()
  const [category, setCategory] = useState<Category | null>(null)
  const [products, setProducts] = useState<Product[]>([])
  const [notFound, setNotFound] = useState(false)

  useEffect(() => {
    // Fetch category
    fetch(`/categories/${slug}`)
      .then(async response => {
        if (!response.ok) {
          setNotFound(true)
          return null
        }
        return response.json()
      })
      .then(data => {
        if (data) {
          setCategory(data)
        }
      })

    // Fetch products for category
    fetch(`/categories/${slug}/products`)
      .then(async response => {
        if (!response.ok) {
          return []
        }
        return response.json()
      })
      .then(data => setProducts(data))
  }, [slug])

  if (notFound) {
    return (
      <section className="category-detail">
        <h1>Categoría no encontrada</h1>
      </section>
    )
  }

  if (!category) {
    return null // Loading state
  }

  return (
    <section className="category-detail">
      <h1>{category.displayName}</h1>
      <div className="category-detail__products-grid">
        {products.map(product => (
          <ProductCard key={product.id} product={product} />
        ))}
      </div>
    </section>
  )
}

export { CategoryDetail }
```

Use `NavLink` instead of `Link` to automatically add classes to active links:

```typescript
import { NavLink } from 'react-router-dom'
import classNames from 'classnames'

<NavLink
  to={`/categories/${category.slug}`}
  className={({ isActive }) =>
    classNames('navigation__link', {
      'navigation__link--active': isActive,
    })
  }
>
  {category.displayName}
</NavLink>
```

The `isActive` property is automatically `true` when the current URL matches the link's `to` prop.

## Implementation Hints

<Accordion title="How do I link to a category?">

Simple approach using string concatenation:

```typescript
to={`/categories/${category.slug}`}
```

Robust approach using `generatePath`:

```typescript
import { generatePath } from 'react-router-dom'
import { PATHS } from '../routes'

to={generatePath(PATHS.CATEGORY_DETAIL, { slug: category.slug })}
```

</Accordion>

<Accordion title="Why check response.ok before parsing JSON?">

If the response is a 404, calling `response.json()` will fail because there's no JSON body:

```typescript
const response = await fetch(`/categories/${slug}`)
if (!response.ok) {
  setNotFound(true)
  return null
}
const data = await response.json()
```

Always check `response.ok` (true for status 200-299) before parsing.

</Accordion>

<Accordion title="How do I navigate to a specific URL in tests?">

Use `window.history.pushState` before rendering:

```typescript
window.history.pushState({}, '', '/categories/fruta-y-verdura')
render(<App />)
```

This simulates the user navigating directly to that URL.

</Accordion>

<Accordion title="How do I filter products by category?">

In the MSW handler, find the category first, then filter products:

```typescript
http.get('/categories/:slug/products', ({ params }) => {
  const { slug } = params
  const category = categoriesFixtures.find(c => c.slug === slug)
  
  if (!category) {
    return new HttpResponse(null, { status: 404 })
  }
  
  const categoryProducts = productsFixtures.filter(
    p => p.categoryId === category.id
  )
  
  return HttpResponse.json(categoryProducts)
})
```

</Accordion>
